iq-select

必须
  listApi: string;                               //下拉列表选项api
  dataModal: any = {{'{'}}{{'}'}};                   //data数据模型
非必需
  placeHolder: string;                     //placeholder
  modalShow: boolean = true;       //弹出框出现
  noSearch: boolean = false;          //隐藏搜索框
  disabled: boolean = false;           //禁用
  queryParams: any = {{'{}'}};                  //额外的查询参数
  required: boolean = false;
  itemIndex: number = 0;              //被选项的位置
  itemValue: number = 0;              //ngModel绑定值的位置
